# useUTXOs

Hook to get the UTXOs of the connected accounts.

## Usage

```tsx
import {useUTXOs} from '@midl-xyz/midl-js-react';

function App () {
    const {utxos, isLoading} = useUTXOs();

    return (
        <div>
            {utxos.map(utxo => (
                <div key={utxo.txid}>{utxo.txid}</div>
            ))}
        </div>
    );
}
```
