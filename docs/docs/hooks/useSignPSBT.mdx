# useSignPSBT

Hook to sign PSBT with the connected accounts.

## Usage 

```tsx
import {useSignPSBT, useAccounts} from '@midl-xyz/midl-js-react';
import {Psbt} from 'bitcoinjs-lib';

function App () {
    const {signPSBT, isPending} = useSignPSBT();
    const {accounts} = useAccounts();
    const psbt = new Psbt();
    // ... Add inputs and outputs to the PSBT

    const paymentAccount = accounts.find(account => account.purpose === AddressPurpose.Payment);
    const ordinalsAccount = accounts.find(account => account.purpose === AddressPurpose.Ordinals);

    // Sign inputs with the corresponding index
    const signInputs = {
        [paymentAccount.address]: [0],
        [ordinalsAccount.address]: [1]
    };

    return (
        <div>
            <button onClick={() => signPSBT({
                psbt: psbt.toBase64(),
                signInputs
            })} disabled={isPending}>
                Sign Message
            </button>
        </div>
    );
}
```

## Returns

- **signPSBT**: `(params: {psbt: string, signInputs: {[address: string]: number[]}}) => Promise<string>` - Function to sign PSBT.
- **signPSBTAsync**: `(params: {psbt: string, signInputs: {[address: string]: number[]}}) => Promise<string>` - Function to sign PSBT asynchronously.
